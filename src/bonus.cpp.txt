#include <Arduino.h>
#include <arduinoFFT.h>
#include <algorithm>

#define ANOMALY_PROB 0.1    // 0.01 (1%), 0.05 (5%), 0.10 (10%)
#define FILTER_WINDOW 15    // 5, 15, 31
#define Z_THRESHOLD_MULTIPLIER 3.0  // 3-sigma rule for Z-Score
#define SAMPLES 128
#define SAMPLING_FREQ 100.0

// FFT Arrays and Object
double vReal[SAMPLES];
double vImag[SAMPLES];
ArduinoFFT<double> FFT = ArduinoFFT<double>(vReal, vImag, SAMPLES, SAMPLING_FREQ);

// Signal Buffers
float clean_signal[SAMPLES];
float raw_signal[SAMPLES];
bool is_anomaly[SAMPLES];
float filtered_signal[SAMPLES];

struct FilterMetrics {
    unsigned long exec_time_us;
    int true_positives;
    int false_positives;
    int total_anomalies;
    float mean_error_reduction;
};


// ------------ 1. SIGNAL GENERATION & DSP UTILS ------------
float getGaussianNoise(float mu, float sigma) {
    float u1 = max(0.0001f, (float)random(10000) / 10000.0f);
    float u2 = max(0.0001f, (float)random(10000) / 10000.0f);
    float z0 = sqrt(-2.0 * log(u1)) * cos(2.0 * PI * u2);
    return z0 * sigma + mu;
}

float calculateMER(float* filtered, float* raw, float* clean, int length) {
    float raw_error = 0.0;
    float filtered_error = 0.0;
    for (int i = 0; i < length; i++) {
        raw_error += abs(raw[i] - clean[i]);
        filtered_error += abs(filtered[i] - clean[i]);
    }
    if (raw_error == 0) return 0.0;
    return ((raw_error - filtered_error) / raw_error) * 100.0;
}

void generateSignalBuffer() {
    randomSeed(micros());
    for (int i = 0; i < SAMPLES; i++) {
        float t = i / SAMPLING_FREQ;
        
        // signal: s(t) = 2*sin(2*pi*3*t) + 4*sin(2*pi*5*t)
        clean_signal[i] = 2.0 * sin(2 * PI * 3 * t) + 4.0 * sin(2 * PI * 5 * t);
        float noise = getGaussianNoise(0.0, 0.2);
        float anomaly_spike = 0.0;
        is_anomaly[i] = false;
        
        if (random(10000) < (ANOMALY_PROB * 10000)) {
            // Random spike between 5 and 15, random sign
            anomaly_spike = random(500, 1500) / 100.0; 
            if (random(2) == 0) anomaly_spike = -anomaly_spike;
            is_anomaly[i] = true;
        }
        raw_signal[i] = clean_signal[i] + noise + anomaly_spike;
    }
}

double getDominantFrequency(float* input_signal) {
    for (int i = 0; i < SAMPLES; i++) {
        vReal[i] = input_signal[i];
        vImag[i] = 0.0;
    }
    FFT.windowing(FFT_WIN_TYP_HAMMING, FFT_FORWARD);
    FFT.compute(FFT_FORWARD);
    FFT.complexToMagnitude();
    return FFT.majorPeak();
}


// ------------ 2. Z-SCORE ------------
FilterMetrics zscoreFilter() {
    FilterMetrics m = {0, 0, 0, 0, 0.0};
    for (int i = 0; i < SAMPLES; i++) {
        if (is_anomaly[i]) m.total_anomalies++;
    }

    int half_window = FILTER_WINDOW / 2;
    unsigned long start_time = micros();

    for (int i = 0; i < SAMPLES; i++) {
        int start_idx = max(0, i - half_window);
        int end_idx = min(SAMPLES - 1, i + half_window);
        int window_size = end_idx - start_idx + 1;

        // Mean
        float sum = 0.0;
        for (int j = start_idx; j <= end_idx; j++) sum += raw_signal[j];
        float mean = sum / window_size;

        // Standard Deviation
        float variance_sum = 0.0;
        for (int j = start_idx; j <= end_idx; j++) {
            float diff = raw_signal[j] - mean;
            variance_sum += (diff * diff);
        }
        float std_dev = sqrt(variance_sum / window_size);

        // Z-Score Detection
        if (abs(raw_signal[i] - mean) > (Z_THRESHOLD_MULTIPLIER * std_dev)) {
            filtered_signal[i] = mean;
            if (is_anomaly[i]) m.true_positives++;
            else m.false_positives++;
        } else {
            filtered_signal[i] = raw_signal[i];
        }
    }
    m.exec_time_us = micros() - start_time;
    m.mean_error_reduction = calculateMER(filtered_signal, raw_signal, clean_signal, SAMPLES);
    return m;
}


// ------------ 3. HAMPEL ------------
FilterMetrics hampelFilter() {
    FilterMetrics m = {0, 0, 0, 0, 0.0};
    for (int i = 0; i < SAMPLES; i++) {
        if (is_anomaly[i]) m.total_anomalies++;
    }

    int half_window = FILTER_WINDOW / 2;
    float window_array[FILTER_WINDOW];
    
    unsigned long start_time = micros();

    for (int i = 0; i < SAMPLES; i++) {
        int start_idx = max(0, i - half_window);
        int end_idx = min(SAMPLES - 1, i + half_window);
        int window_size = end_idx - start_idx + 1;

        for (int j = 0; j < window_size; j++) {
            window_array[j] = raw_signal[start_idx + j];
        }

        // Median
        std::sort(window_array, window_array + window_size);
        float median = window_array[window_size / 2];

        // Median Absolute Deviation (MAD)
        float mad_array[FILTER_WINDOW];
        for (int j = 0; j < window_size; j++) {
            mad_array[j] = abs(raw_signal[start_idx + j] - median);
        }
        std::sort(mad_array, mad_array + window_size);
        float mad = mad_array[window_size / 2];
        float threshold = 3.0 * mad;

        // Hampel Detection
        if (abs(raw_signal[i] - median) > threshold) {
            filtered_signal[i] = median;
            if (is_anomaly[i]) m.true_positives++;
            else m.false_positives++;
        } else {
            filtered_signal[i] = raw_signal[i];
        }
    }
    m.exec_time_us = micros() - start_time;
    m.mean_error_reduction = calculateMER(filtered_signal, raw_signal, clean_signal, SAMPLES);
    return m;
}


void setup() {
    Serial.begin(115200);
    delay(2000);
}


void loop() {
    Serial.println("\n========================================");
    Serial.printf("Testing at P=%.2f, WINDOW=%d\n", ANOMALY_PROB, FILTER_WINDOW);
    
    generateSignalBuffer();

    double peak_unfiltered = getDominantFrequency(raw_signal);
    Serial.printf("[UNFILTERED] FFT Peak: %.2f Hz\n", peak_unfiltered);

    FilterMetrics z = zscoreFilter();
    double peak_z = getDominantFrequency(filtered_signal);
    float z_tpr = 0.0;
    if (z.total_anomalies > 0) 
        z_tpr = (float)z.true_positives / z.total_anomalies;
    int clean_samples_z = SAMPLES - z.total_anomalies;
    float z_fpr = 0.0;
    if (clean_samples_z > 0)
        z_fpr = (float)z.false_positives / clean_samples_z;
    Serial.printf("[Z-SCORE] Exec: %lu us | TPR: %.2f | FPR: %.2f | MER: %5.1f%% | FFT Peak: %.2f Hz\n", z.exec_time_us, z_tpr, z_fpr, z.mean_error_reduction, peak_z);

    FilterMetrics h = hampelFilter();
    double peak_h = getDominantFrequency(filtered_signal);
    float h_tpr = 0.0;
    if (h.total_anomalies > 0)
        h_tpr = (float)h.true_positives / h.total_anomalies;
    int clean_samples_h = SAMPLES - h.total_anomalies;
    float h_fpr = 0.0;
    if (clean_samples_h > 0)
        h_fpr = (float)h.false_positives / clean_samples_h;
    Serial.printf("[HAMPEL]  Exec: %lu us | TPR: %.2f | FPR: %.2f | MER: %5.1f%% | FFT Peak: %.2f Hz\n", h.exec_time_us, h_tpr, h_fpr, h.mean_error_reduction, peak_h);

    delay(3000);
}
